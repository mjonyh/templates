%% Preamble Configuration: preamble/packages.tex

% --- Core Geometry & Spacing ---
% Left margin is 1.5 inches for binding clearance; others are 1.0 inch
\usepackage[top=1.0in, bottom=1.0in, left=1.5in, right=1.0in, headheight=15pt]{geometry}
\usepackage{setspace}
\onehalfspacing % Default to 1.5 spacing for front matter

% --- Fonts & Encoding ---
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{lmodern} % Computer Modern font family (default)

% To switch to Times New Roman, uncomment the following line:
% \usepackage{newtxtext,newtxmath}

% --- Navigation & Headers/Footers ---
\usepackage{fancyhdr}
\usepackage{emptypage} % Ensure empty pages have no headers/footers
\usepackage{tocbibind}  % Automatically adds bibliography, lists, index to TOC

% Configure fancy header style
\pagestyle{fancy}
\fancyhf{} % Clear default headers and footers

\makeatletter
\if@twoside
  \fancyhead[LE,RO]{\slshape\small\nouppercase{\rightmark}} % Section title
  \fancyhead[LO,RE]{\slshape\small\nouppercase{\leftmark}}  % Chapter title
\else
  \fancyhead[L]{\slshape\small\nouppercase{\leftmark}}      % Only print Chapter title on left
\fi
\makeatother
\fancyfoot[R]{\thepage}               % Right footer: Page number
\renewcommand{\headrulewidth}{0.4pt}  % Header underline width
\renewcommand{\footrulewidth}{0.0pt}

% Redefine plain page style (used on Chapter openings and front matter)
\fancypagestyle{plain}{
    \fancyhf{}
    \fancyfoot[C]{\thepage}           % Center page number on plain pages
    \renewcommand{\headrulewidth}{0.0pt}
}

% --- Chapter & Section Formatting ---
\usepackage{titlesec}
\titleformat{\chapter}[display]
  {\normalfont\large\bfseries\centering}
  {\chaptertitlename\ \thechapter}
  {10pt}
  {\Large}

% --- Graphics & Visuals ---
\usepackage{graphicx}

% --- Math & Sciences ---
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{siunitx}
\sisetup{
  scientific-notation = false,
  round-mode = places,
  round-precision = 3,
  exponent-product = \times
}

\DeclareSIUnit\mmHg{\milli\meter\tothe{-2}\tothe{-1}\of{Hg}}
