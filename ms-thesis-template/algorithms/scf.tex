\begin{algorithm}[H]
\SetAlgoLined
\KwIn{Initial trial electron density $\rho_0(\mathbf{r})$; mixing parameter
$\beta \in [0,1]$; convergence threshold $\eta$; maximum iterations $N_{\max}$}
\KwOut{Converged ground-state density $\rho(\mathbf{r})$ and total energy $E_{\text{tot}}$}

$\rho \leftarrow \rho_0$\;
$k \leftarrow 0$\;
\Repeat{convergence or $k = N_{\max}$}{
  Construct effective potential $V_{\text{eff}}(\mathbf{r})$ from Eq.~(N.M.2)\;
  Solve Kohn-Sham equations for $\{\psi_i\}$\;
  Compute new density $\rho_{\text{new}}(\mathbf{r}) \leftarrow
      \sum_i f_i \, |\psi_i(\mathbf{r})|^2$\;
  Mix densities: $\rho \leftarrow \rho + \beta (\rho_{\text{new}} - \rho)$\;
  Compute $E_{\text{tot}}$\;
  $k \leftarrow k + 1$\;
}
\caption{Self-consistent field (SCF) iteration for Kohn-Sham DFT.}
\label{alg:scf}
\end{algorithm}
